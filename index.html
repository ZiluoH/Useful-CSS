  4. A bouncing-loader

                        
                        @keyframes bouncing-loader{
                            from {
                                opacity: 1;
                                transform: translateY(0);
                            }
                            to {
                                opacity: 0.1;
                                transform: translateY(-1rem);
                            }
                        }

                        #eg4 .bouncing-loader{
                            display: flex;
                            justify-content: center;
                        }

                        #eg4 .bouncing-loader > div {
                            width: 1rem;
                            height: 1rem;
                            margin: 3rem 0.2rem;
                            background: #8385aa;
                            border-radius: 50%;
                            animation: bouncing-loader 0.6s infinite alternate;
                        }

                        #eg4 .bouncing-loader div:nth-child(2){
                            animation-delay: 0.2s;
                        }

                        #eg4 .bouncing-loader div:nth-child(3){
                            animation-delay: 0.4s;
                        }
